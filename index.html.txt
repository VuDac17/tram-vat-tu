<script>
const FIREBASE_URL = "https://khonoithat-38dd7-default-rtdb.firebaseio.com/";
const WORKER_PIN = "5678";
let thongTinVatTuHienTai = null;

function moKhoaTramVatTu() {
    let pin = document.getElementById("input-pin-tho").value;
    if(pin === WORKER_PIN) {
        document.getElementById("man-hinh-pin").style.display = "none";
        document.getElementById("app-chinh").style.display = "block";
        taiDanhSachCongTrinh();
    } else {
        Swal.fire("Sai mã PIN", "Mã PIN công nhân kho không đúng!", "error");
    }
}

async function taiDanhSachCongTrinh() {
    try {
        let res = await fetch(`${FIREBASE_URL}du_an.json`);
        let data = await res.json();
        let select = document.getElementById("select-du-an");
        if(data) {
            for(let key in data) {
                let opt = document.createElement("option");
                opt.value = key;
                opt.innerText = data[key].ten_du_an;
                select.appendChild(opt);
            }
        }
    } catch(e){}
}

async function truyVanVatTuGoc() {
    let ma = document.getElementById("input-ma-vt").value.trim().toUpperCase();
    if(!ma) { document.getElementById("khung-xem-truoc").style.display = "none"; return; }
    try {
        let res = await fetch(`${FIREBASE_URL}kho_vat_tu/${ma}.json`);
        let data = await res.json();
        if(data) {
            thongTinVatTuHienTai = data;
            document.getElementById("lbl-ten").innerText = data.ten_vt;
            document.getElementById("lbl-gia").innerText = Number(data.don_gia).toLocaleString('vi-VN');
            document.getElementById("lbl-ton").innerText = data.so_luong;
            document.getElementById("khung-xem-truoc").style.display = "block";
        } else {
            document.getElementById("khung-xem-truoc").style.display = "none";
            thongTinVatTuHienTai = null;
        }
    } catch(e){}
}

async function thucHienGiaoDich(loai) {
    let ma = document.getElementById("input-ma-vt").value.trim().toUpperCase();
    let sl = parseInt(document.getElementById("input-sl").value);
    let ma_da = document.getElementById("select-du-an").value;
    
    if(!thongTinVatTuHienTai) {
        Swal.fire("Lỗi mã", "Mã vật tư này chưa được văn phòng định giá gốc!", "warning");
        return;
    }
    if(!sl || sl <= 0) {
        Swal.fire("Lỗi số lượng", "Vui lòng nhập số lượng hợp lệ!", "warning");
        return;
    }
    
    let ton_cu = thongTinVatTuHienTai.so_luong || 0;
    let ton_moi = loai === "NHẬP KHO" ? (ton_cu + sl) : (ton_cu - sl);
    
    if(loai === "XUẤT KHO" && ton_cu < sl) {
        Swal.fire("Không đủ hàng", `Kho chỉ còn tồn thực tế ${ton_cu}!`, "error");
        return;
    }
    
    let d = new Date();
    let ngay_ghi = d.toLocaleDateString('vi-VN', {day: '2-digit', month: '2-digit', year: 'numeric'});
    let gio_ghi = d.toLocaleTimeString('vi-VN', {hour12: false});
    
    try {
        await fetch(`${FIREBASE_URL}kho_vat_tu/${ma}/so_luong.json`, { method: 'PUT', body: JSON.stringify(ton_moi) });
        let id_ls = "LS_" + Date.now();
        await fetch(`${FIREBASE_URL}nhat_ky_nhap_xuat/${id_ls}.json`, {
            method: 'PUT',
            body: JSON.stringify({
                ma_vt: ma, loai: loai, so_luong: sl, don_gia: thongTinVatTuHienTai.don_gia,
                ma_du_an: ma_da, ngay: ngay_ghi, gio: gio_ghi
            })
        });
        
        Swal.fire("Thành công", `Đã lưu giao dịch ${loai}!`, "success");
        document.getElementById("input-ma-vt").value = "";
        document.getElementById("input-sl").value = "1";
        document.getElementById("khung-xem-truoc").style.display = "none";
        thongTinVatTuHienTai = null;
    } catch(e) {
        Swal.fire("Lỗi", "Mất kết nối mạng xưởng!", "error");
    }
}
